# Sample curl commands for testing your URL Shortener Microservice

# 1. POST a URL to /api/shorturl
curl -X POST -H "Content-Type: application/x-www-form-urlencoded" -d "url=https://freeCodeCamp.org" http://localhost:3000/api/shorturl

# Expected response:
# { "original_url": "https://freeCodeCamp.org", "short_url": 1 }

# 2. GET /api/shorturl/1 (replace 1 with the short_url from the previous response)
curl -I http://localhost:3000/api/shorturl/1

# Expected response:
# HTTP/1.1 302 Found
# Location: https://freeCodeCamp.org
